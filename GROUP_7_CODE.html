# Function to get recommendations based on user ratings
def recommend_books(book_ratings, n_recommendations=5):
    """
    Generate book recommendations based on user ratings.

    Args:
        book_ratings: List of (book_title, rating) tuples
        n_recommendations: Number of recommendations to return

    Returns:
        DataFrame of recommendations, list of book details for explanations
    """
    if not book_ratings:
        return pd.DataFrame([{"Message": "Please rate at least one book first"}]), []

    try:
        new_user_id = ratings_df['User-ID'].max() + 1
        new_ratings = []
        for book_name, rating in book_ratings:
            if not book_name or rating <= 0:
                continue
            book_row = books_df[books_df['Book-Title'] == book_name]
            if book_row.empty:
                continue
            isbn = book_row.iloc[0]['ISBN']
            new_ratings.append({
                'User-ID': new_user_id,
                'ISBN': isbn,
                'Book-Rating': rating
            })

        if not new_ratings:
            return pd.DataFrame([{"Message": "Could not find any of the specified books"}]), []

        new_ratings_df = pd.DataFrame(new_ratings)
        all_ratings = pd.concat([ratings_df, new_ratings_df])
        new_data = Dataset.load_from_df(all_ratings[['User-ID', 'ISBN', 'Book-Rating']], Reader(rating_scale=(1, 10)))
        new_trainset = new_data.build_full_trainset()
        new_model = SVD()
        new_model.fit(new_trainset)

        rated_books = set(new_ratings_df['ISBN'].values)
        all_books = set(books_df['ISBN'].values)
        unrated_books = list(all_books - rated_books)

        predictions = []
        for isbn in unrated_books:
            est = new_model.predict(new_user_id, isbn).est
            predictions.append((isbn, est))

        predictions.sort(key=lambda x: x[1], reverse=True)
        top_n = predictions[:n_recommendations]

        results = []
        book_details = []
        for isbn, est in top_n:
            book_info = books_df[books_df['ISBN'] == isbn].iloc[0]
            results.append({
                'Book': book_info['Book-Title'],
                'Author': book_info['Book-Author'],
                'Predicted Rating': round(est, 2)
            })
            book_details.append({
                'title': book_info['Book-Title'],
                'author': book_info['Book-Author'],
                'predicted_rating': est
            })

        return pd.DataFrame(results), book_details

    except Exception as e:
        return pd.DataFrame([{"Message": f"Error: {str(e)}"}]), []

# Function to generate explanation using OpenAI
def generate_explanation(client, book_title, book_author, predicted_rating, user_ratings):
    """
    Generate an explanation for why a book was recommended using OpenAI's GPT model.

    Args:
        client: OpenAI client instance
        book_title: Title of the recommended book
        book_author: Author of the recommended book
        predicted_rating: Predicted rating for the book
        user_ratings: List of (book_title, rating) tuples rated by the user

    Returns:
        str: Explanation for the recommendation
    """
    try:
        ratings_text = ""
        for title, rating in user_ratings:
            if title and rating > 0:
                ratings_text += f"- {title}: {rating}/10\n"

        prompt = f"""
As a book recommendation system, explain why the book "{book_title}" by {book_author}
with a predicted rating of {predicted_rating:.2f}/10 would be recommended to a user
who has rated the following books:

{ratings_text}

Give a friendly, conversational explanation that highlights:
1. Connections to the user's taste based on their ratings
2. Author or genre preferences the user might have
3. Specific elements of the recommended book that align with their preferences
4. Any interesting insights about this recommendation

Keep the explanation concise, around 3-4 sentences.
"""
        response = client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[
                {"role": "system", "content": "You are a helpful book recommendation assistant."},
                {"role": "user", "content": prompt}
            ],
            max_tokens=500,
            temperature=0.7
        )
        return response.choices[0].message.content
    except Exception as e:
        return f"I couldn't generate an explanation at this time. Error: {str(e)}"

def explain_recommendation(client, selected_book, user_ratings, book_details):
    """
    Generate explanation for a selected book.

    Args:
        client: OpenAI client instance
        selected_book: Title of the selected book
        user_ratings: List of user ratings
        book_details: List of recommended book details

    Returns:
        str: Explanation text
    """
    if not client:
        return "OpenAI client not properly initialized."
    if not selected_book or not user_ratings or not book_details:
        return "Please select a book from the recommendations first."

    selected_book_info = None
    for book in book_details:
        if book['title'] == selected_book:
            selected_book_info = book
            break

    if not selected_book_info:
        return "Book information not found."

    explanation = generate_explanation(
        client,
        selected_book_info['title'],
        selected_book_info['author'],
        selected_book_info['predicted_rating'],
        user_ratings
    )
    return explanation
